#import the pyplot and wavfile modules 

import matplotlib.pyplot as plot

from scipy.io import wavfile

 

# Read the wav file (mono)
wave_path = 'C:/Users/bomnyoung/Downloads/Berlin emo DB/wav/10a02Ab.wav'

samplingFrequency, signalData = wavfile.read(wave_path)

 

# Plot the signal read from wav file

plot.subplot(211)

plot.title('Spectrogram of a wav file')

 

plot.plot(signalData)

plot.xlabel('Sample')

plot.ylabel('Amplitude')

 

plot.subplot(212)

test=plot.specgram(signalData,Fs=samplingFrequency)

plot.xlabel('Time')

plot.ylabel('Frequency')

 

plot.show()
print(test)
